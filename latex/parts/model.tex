\section*{Exponential growth ODE}

\begin{align*}
    \frac{dN}{dt} = r N,
\end{align*}


\section*{Solution}

\begin{align*}
    N = A e^{r t},
\end{align*}

\section*{Logistic differential equation }

\begin{align*}
    \frac{dN}{dt} = r N \Big(1 - \frac{N}{K} \Big),
\end{align*}


\section*{Solution of logistic differential equation }

\begin{align*}
    N = \frac{K}{Q e^{-r t} + 1},
\end{align*}
\begin{align*}
    Q = \frac{K}{A} - 1.
\end{align*}
